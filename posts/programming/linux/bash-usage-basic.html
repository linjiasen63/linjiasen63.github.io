2. 文件处理命令

2.1 目录命令

ls

# -l --long的简写，展示具体信息
# -a --all的简写，展示所有内容
# -d 展示指定文件夹的信息
# -i 输出文件在系统中唯一的id信息
# -h 人性化输出（如将文件大小自动转换为KB、MB等）

mkdir

# 全称：make directories

# -p 创建指定路径中不存在的目录
# 其中tmp目录不存在
mkdir -p ./tmp/200606

cd

# 全称：change directories
# cd 指定路径
cd ..
cd /root
cd ./tmp/200606

pwd

# 全称：print working directories
# 输出当前目录
pwd

rmdir

# 全称：remove empty directories
# 删除空目录

rm

# 全称：remove empty
# 删除文件
# -r 删除目录
# -f 强制删除，不会二次询问

cp

# -r 拷贝目录
cp -r ./tmp/200606 ./

mv

# 移动与重命名
mv ./tmp/200606 ./2006

2.2 文件命令

# 退出：q

touch、echo

# touch 文件名
touch 1.txt
# 将内容保存到文件
echo '1' > 1.txt
# 将内容追加到文件
echo '2' >> 1.txt

cat、tac

# -n 显示行号
# 从前往后输出内容
cat 1.txt
# 从后往前输出内容
tac 1.txt

more、lesss

# 从前往后输出内容

# more与less共用
#   下一行：enter
#   下一页：space或f

# less
#   上一行：向上简头
#   下一行：向下简头
#   上一页：PgUp
#   下一页：PgDn

head、tail

# 读取顶部几行，其中5为具体的行数
head 5 /etc/nginx/nginx.conf
# 读取前几行，其中5为具体的行数
# -f 读取后不退出，当文件数据变化时自动刷新
tail 5 /etc/nginx/nginx.conf

3. 权限管理命令

3.1 权限说明

权限名称描述文件目录
r读权限可以查看文件内容可以列出目录中的内容
w写权限可以修改文件内容可以在目录中创建、删除文件
x执行权限可以执行文件可以进入目录

3.2 相关命令

chmod

# 全称： change the permissions mode of a file
# 语法： chmod [{ugoa}{+-=}{rwx}] [文件或目录]
#             [mode=421] [文件或目录]
#             -R 递归修改

chown

# 描述：改变文件或目录的所有者
# 全称：change file ownership
# 语法：chown [用户] [文件或目录]
chown linjs 1.txt

chgrp

# 描述：改变文件或目录的所属组
# 全称：change file group ownership
# 语法：chown [用户组] [文件或目录]
chown devp 1.txt

umask

# 描述：显示、设置文件的缺省权限
# 全称：the user file-creation mask
# 语法：umask [-S]
#      -S 以rwx形式显示新建文件的缺省权限

# 输出 u=rwx,g=rx,o=rx
# 注意目录是完全按照缺省权限来的，但文件的所有者、所属组与其他人没有可执行权限x
umask -S 
# 输出 022 
# 与 077 进行与运算得到真实权限：755
umask 

# 设置缺省权限700，需要执行以下命令
umask 077

4. 文件查找命令

find

# 功能：文件搜索
# 语法：find [搜索范围] [匹配条件]

# 文件名称查找
# -name 严格区分大小写
find /etc -name nginx
# -iname 不区分大小写
find /etc -iname nginx 
# 模糊匹配
# * 匹配多个字符
# ? 匹配一个字符
find /etc -name 'nginx*'
find /etc -name '?.txt'

# 文件大小查找
# -size +201800
# +n 大于；-n 小于；n 等于
find / -size +204800

# 根据所有者查找
# -user [所有者]
find / -user root

# 根据所属组查找
# -group [所属组]
find / -group devp

# 根据时间进行查找
# -amin 访问时间access
# -cmin 文件属性change
# -mmin 文件内容modify

# -n 多少分钟内；+n 多少分钟外；n 多少分钟

# 在/etc下查找5分钟内被修改过属性的文件和目录
find /etc -cmin -5

# 条件组件查询
# -a 两个条件同时满足
# -o 两个条件满足任意一个即可

# 对搜索结果执行
# -exec/-ok 命令 {} \;
# -ok命令会询问每个查询子记录操作是否执行，而-exec则不会
# 查找当前目录下文件名为1.txt的文件，然后删除
find . -name 1.txt -a -type f -exec rm {} \;

# 根据文件类型查找
# -type
# f 文件；d 目录；l 软链接文件
find . -type f

# 根据i节点查找
# -inum [i节点编号]
find . -inum 530860

linux系统以512字节大小为一个数据块，因此 1相当于0.5MB，根据此换算法则：100MB相当于204800

locate

用于查询系统文件，linux会定时将系统的文件信息更新到数据库中（但tmp文件夹的内容不会），同时用户可以使用updatedb更新文件信息

# locate 文件名称

which

# 查询命令所在的位置与别名信息
which ll

whereis

# 查询命令所在的位置与帮助文件
whereis ll

grep

# 作用：在文件中搜索内容
# 语法：grep 搜索内容 文件名称
#      -i 忽略大小写
#      -v 剔除符合的部分

# 输出文件中不包含注释的内容
grep -v '^#' /etc/nginx/nginx.conf

5. 帮助命令

man

# 功能：获取帮助信息
# 全称：manual
# 语法：man [命令或配置文件]

# 查看ls命令的帮助信息
man ls
# 查看配置文件services的帮助信息
man services

# 如果 命令或配置文件同名，可使用标记区分
# 1 命令的帮助
# 5 配置文件的帮助

# 操作与more命令雷同
# 下一行：enter
# 下一页：space或f
# 退出：q
# 搜索选项：/-选择参数，如 /-l

whatis

# 功能：显示命令功能描述
# whatis 命令名称
whatis ls

apropos

# 功能：显示配置文件描述
# apropos 配置名称
apropos services

help

# 功能：获得shell内置命令的帮助信息
# shell内置命令，某些shell不包含
help umask

7. 压缩解压命令

gzip、gunzip

# gzip
# 功能：压缩文件（生成压缩文件后会删除原文件），只能操作文件
# 原意：GNU zip
# 格式：.gz
gzip 1.txt # 生成 1.txt.gz

# gunzip
# 功能：解压文件（生成解压文件后会删除原文件）
gunzip 1.txt.gz  #生成 1.txt
gzip -d 1.txt.gz  
#后缀名需要为.gz

tar

# 压缩解压命令
# 功能：打包目录
# 语法：tar 选项[-cvf] [压缩后文件名] [目录]
#      -c 打包
#      -v 显示详细信息
#      -f 指定文件名
#      -z 打包同时使用zip压缩
#      -j 打包同时使用bzip2压缩
# 格式：.tar
tar -cvf nginx.tar nignx
# 格式：.tar.gz
tar -zcvf nginx.tar.gz nginx
# 格式：.tar.bz2
tar -jcvf nginx.targ.bz2 nginx

# 功能：解压目录
# 语法：tar 选项[-xvf] [压缩后文件名]
#      -x 解压
#      -v 显示详细信息
#      -f 指定文件名
#      -z 拆包同时使用zip解压
#      -j 拆包同时使用bzip2解压
# 格式：.tar
tar -xvf nginx.tar
# 格式：.tar.gz
tar -zxvf nginx.tar.gz
# 格式：.tar.bz2
tar -jxvf nginx.tar.bz2

zip、unzip

# zip
# 功能：压缩文件或目录
# 语法：zip 选项[-r] [压缩后文件名] [文件或目录]
#      -r 压缩目录
# 格式：.zip
zip 1.txt.zip 1.txt
zip -r nginx.zip nginx

# unzip
# 功能：解压文件或目录
unzip 1.txt.zip
unzip nginx.zip

bzip2、bunzip2

# 功能：压缩文件
# 语法：bzip2 选项[-k] [文件]
#      -k 压缩后保留原文件
# 格式：.bz
bzip2 -k 1.txt

# 功能：解压文件或
# 语法：bunzip2 选项[-k] [压缩后文件名]
#      -k 解压后保留原文件
bunzip2 -k 1.txt.bz

8. 网络操作命令

write、wall

# write
# 功能：给用户发消息，以Ctrl+D保存结束
# 语法：wirte <用户名>
write linjs

# wall
# 原意：write all
# 功能：发广播信息
# 语法：wall [message]

ping、ifconfig

# ping
# 功能：测试网络连通性
# 语法：ping 选项 IP地址
#      -c 指定发送次数
ping baidu.com

# ifconfig
# 原意：interface configure
# 功能：查看和设置网卡信息
# 权限：root
ifconfig
ifconfig etho 192.168.1.110

mail

# 功能：查看发送电子邮件
# 语法：mail [用户名]

# 发送（输入Subject回车后，输入内容，最后Ctrl+D发送）
mail linjs

# 查看
mail # 查看所有邮件，并进入编辑模板（输入help查看帮助）
# 输入邮件编号，查看内容（返回输入h）

last、lastlog

# last
# 功能：列出目录与过去登入系统的用户信息
last

# lastlog
# 检查某特定用户上次登录的时间
lastlog
last -u 502 # 502为指定用户的id

traceroute

# 显示数据包到主机间的路径
traceroute baidu.com

netstat

# 功能：查看本机网络状态
# 选项：-t：TCP协议
#      -u：UDP协议
#      -l：监听
#      -r：路由
#      -n：显示ip地址和端口号

# 范例
netstat -tuln # 查看本机监听的端口
netstat -an # 查看本机所有的网络连接
netstat -rn # 查看本机路由表

setup

# redhat系统独有
# 功能：配置网络

mount、unmount

# mount
# 语法：mount [-t 文件系统] 设备文件名 挂载点
mount -t iso9660 /dev/sr0 /mnt/cdrom

# unmount
# 当前工作目录在挂载点时，不能执行以下命令（会报busy）
unmount /dev/sr0
unmount /mnt/cdrom
