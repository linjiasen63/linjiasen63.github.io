1. 冒泡排序

bubbleSort

时间复杂度：O(n^2)

function sort(arr) {
  for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr.length - 1 - i; j++) {
      if (arr[j] > arr[j + 1]) {
        const tmp = arr[j + 1];
        arr[j + 1] = arr[j];
        arr[j] = tmp;
      }
    }
  }
}

const arr = [10, 9, 22, -2, -4, 9];
console.log("排序前：", arr);
sort(arr);
console.log("排序后：", arr);

2. 选择排序

selectSort

时间复杂度：O(n^2)

function sort(arr) {
  for (let i = 0; i < arr.length; i++) {
    let index = i;
    for (let j = i + 1; j < arr.length; j++) {
      if (arr[j] < arr[index]) {
        index = j;
      }
    }
    if (index != i) {
      const tmp = arr[i];
      arr[i] = arr[index];
      arr[index] = tmp;
    }
  }
}

const arr = [10, 9, 22, -2, -4, 9];
console.log("排序前：", arr);
sort(arr);
console.log("排序后：", arr);

3. 插入排序

insertSort

时间复杂度：O(n^2)

function sort(arr) {
  for (let i = 1; i < arr.length; i++) {
    if (arr[i] < arr[i - 1]) {
      const tmp = arr[i];
      let j = 0;
      for (j = i - 1; j >= 0 && tmp < arr[j]; j--) {
        arr[j + 1] = arr[j];
      }
      arr[j + 1] = tmp;
    }
  }
}
const arr = [10, 9, 22, -2, -4, 9];
console.log("排序前：", arr);
sort(arr);
console.log("排序后：", arr);

4. 希尔排序

shellSort

时间复杂度与增量(即，步长gap)的选取有关。例如，当增量为1时，希尔排序退化成了直接插入排序，此时的时间复杂度为O(N²)，而Hibbard增量的希尔排序的时间复杂度为O(N3/2)。

function sort(arr) {
  const len = arr.length;
  let increasement = len;
  let i, j, k;
  do {
    increasement = Math.floor(increasement / 3) + 1;
    for (i = 0; i < increasement; i++) {
      for (j = i + increasement; j < len; j++) {
        if (arr[j] < arr[j - increasement]) {
          const tmp = arr[j];
          for (
            k = j - increasement;
            k >= 0 && tmp < arr[k];
            k -= increasement
          ) {
            arr[k + increasement] = arr[k];
          }
          arr[k + increasement] = tmp;
        }
      }
    }
  } while (increasement > 1);
}

const arr = [10, 9, 22, -2, -4, 9];
console.log("排序前：", arr);
sort(arr);
console.log("排序后：", arr);

5. 快速排序

function sort(arr, start, end) {
  if (start >= end) return;
  let i = start;
  let j = end;
  let base = arr[i];
  while (i < j) {
    while (i < j && arr[j] >= base) {
      j--;
    }
    if (i < j) {
      arr[i] = arr[j];
      i++;
    }
    while (i < j && arr[i] < base) {
      i++;
    }
    if (i < j) {
      arr[j] = arr[i];
      j--;
    }
  }
  arr[i] = base;
  sort(arr, start, i);
  sort(arr, i + 1, end);
}

const arr = [10, 9, 22, -2, -4, 9];
console.log("排序前：", arr);
sort(arr, 0, arr.length - 1);
console.log("排序后：", arr);
