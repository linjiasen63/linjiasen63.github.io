es5及之前创建对象的常见模式。参考书籍：《javascript高级程序设计》（第3版）

1. 工厂模式

  • 优点：解决创建多个相似对象的问题
  • 缺点：没有解决对象识别的问题（即怎样知道对象的类型）
function createObj(name) {
  var o = new Object();
  o.name = name;
  o.sayName = function () {
    console.log(this.name);
  };
  return o;
}
// 用法
var obj = createObj('obj');

2. 构造函数模式

  • 优点：解决对象识别的问题（即怎样知道对象的类型）
  • 缺点：各个方法都需要在各个实例上重新创建一遍
function Obj(name) {
  this.name = name;
  this.sayName = function() {
    console.log(this.name);
  };
}
// 用法
var obj = new Obj('obj');

3. 原型模式

  • 优点：解决对象识别的问题（即怎样知道对象的类型）
  • 缺点：各个实例都共用所有属性与方法
function Obj (name) {}
  Obj.prototype.name = name;
  Obj.prototype.sayName = function () {
    console.log(this.name);
  };
}
// 用法
var obj = new Obj('obj');

4. 组合模式

  • 优点：解决对象识别的问题（即怎样知道对象的类型）
  • 缺点：对象的属性定义与方法定义不在一起
function Obj(name) {}
 this.name = name;
}
Obj.prototype.sayName = function () {
  console.log(this.name);
};
// 用法
var obj = new Obj('obj');

5. 动态原型模式

  • 优点：解决对象识别的问题（即怎样知道对象的类型）
  • 不足：没有
function Obj(name) {
 this.name = name;
 if (typeof this.sayName != 'function') {
   Obj.prototype.sayName = function () {
     console.log(this.name);
   };
 }
}
// 用法
var obj = new Obj('obj');

6. 其它

6.1 寄生构造函数模式

  • 优点：
  • 不足：
function CreateObj(name) {
 var o = new Object();
 o.name = name;
 o.sayName = function () {
   console.log(this.name);
 };
 return o;
}
// 用法
var obj = new CreateObj('obj');

6.2 稳妥构造函数模式

  • 优点：用于不支持this以及new关键字的环境
  • 不足：各个方法都需要在各个实例上重新创建一遍
function createObj(name) {
 var o = {};
 o.name = name;
 o.sayName = function () {
   console.log(o.name);
 };
 return o;
}
// 用法
var obj = createObj('obj');
