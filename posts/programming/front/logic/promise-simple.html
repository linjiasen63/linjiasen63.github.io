1. 简单Promise实现

// 1. 简单实现：
class PromiseCustom {
  constructor(fn) {
    this.resolvedCallbacks = [];
    this.rejectedCallbacks = [];
    this.state = 'PENDING';
    this.value = '';
    fn(this.resolve.bind(this), this.reject.bind(this));
  }

  resolve(value) {
    if (this.state === 'PENDING') {
      this.state = 'RESOLVED';
      this.value = value;
      this.resolvedCallbacks.forEach((cb) => cb(this.value));
    }
    return this;
  }

  reject(value) {
    if (this.state === 'REJECTED') {
      this.state = 'REJECTED';
      this.value = value;
      this.rejectedCallbacks.forEach((cb) => cb(this.value));
    }
  }

  then(resolve = function () { }, reject = function () { }) {
    if (this.state === 'PADDING') {
      this.resolvedCallbacks.push(resolve);
      this.rejectedCallbacks.push(reject);
    } else if (this.state === 'RESOLVED') {
      resolve(this.value);
    } else if (this.state === 'REJECTED') {
      reject(this.value);
    }
  }
}

// 2. 用法示例：
const p = new PromiseCustom(function (resolve, reject) {
  console.log('constructor');
  setTimeout(() => {
    return resolve(1);
  }, 1000);
});
p.then(value => {
  console.log('then:' + value);
  return;
})

2. async/await

async function sleep(duration) {
  await new Promise((resolve) => {
    setTimeout(resolve, duration);
  });
}

async function printDelay(msg, delay) {
  await sleep(delay);
  console.log(msg);
}

printDelay('你好，世界！', 3000);

await需要等待promise链接执行结束后，才会继续执行后面的代码

async function sleep(duration) {
  await new Promise((resolve) => {
    console.log('----------- 1');
    setTimeout(resolve, duration);
  }).then(() => {
    console.log('----------- 2');
    return new Promise((resolve) => {
      setTimeout(resolve, duration);
    })
  });
}

async function printDelay(msg, delay) {
  await sleep(delay);
  console.log(msg);
}

printDelay('你好，世界！', 1000);

// 每隔1s输出以下内容
// ----------- 1
// ----------- 2
// 你好，世界！

3. done与finally

Promise.prototype.done = function(resolve, reject) {
  this.then(resolve, rejct)
    .catch(function(reason) {
      setTimeout(() => { throw reason }, 0);
    });
}

Promise.prototype.finally = function(cb) {
  const P = this.constructor;
  return this.then(
    value => P.resolve(callback()).then(() => value),
    reason => P.resolve(callback()).then(() => { throw reason })
  )
}
